const R = require('ramda');
 
// tutorial 1: add 1 to all elements of an array (map+add)
const data1 = [1, 2, 3, 4];
const add1 = R.add(1);
const tutorial1 = R.map(add1);
console.log('tutorial1:', tutorial1(data1));
 
// tutorial 2: sum of all elements of an array (reduce+add)
const data2 = [1, 2, 3];
const tutorial2 = R.reduce(R.add, 0);
console.log('tutorial2:', tutorial2(data2));
 
// tutorial3: factorial of all elements of an array (reduce+multiply+times)
const data3 = [1, 2, 3];
const tutorial3a = R.reduce(R.multiply, 1);
console.log('tutorial3a:', tutorial3a(data3));
 
let createArray = size => Array.from(new Array(size), (x, i) => i + 1);
const tutorial3b = size => R.reduce(R.multiply, 1, createArray(size));
console.log('tutorial3b:', tutorial3b(3));
 
createArray = size => R.times(add1)(size);
const tutorial3c = size => R.reduce(R.multiply, 1, createArray(size));
console.log('tutorial3c:', tutorial3c(3));
 
// tutorial4: Transform data as an Objet with { [data[i].id]: data[i] } (map+fromPairs)
const data4 = [
  { id: 1, label: 'data1' },
  { id: 2, label: 'data2' },
  { id: 3, label: 'data3' },
  { id: 4, label: 'data4' },
];
const tutorial4 = data => R.fromPairs(R.map(d => [d.id, d], data));
console.log('tutorial4', '\n', tutorial4(data4));
